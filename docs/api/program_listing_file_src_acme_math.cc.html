  real
  clamp(
      real input,
      real low,
      real high)
  {
    if (input < low)
      return low;
    else if (input > high)
      return high;
    else
      return input;
  }

  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -

  angleaxis
  rotate(
      real angle,
      std::string const &axis)
  {
    if (axis == "X")
      return Eigen::AngleAxis<real>(angle, UnitX_vec3);
    else if (axis == "Y")
      return Eigen::AngleAxis<real>(angle, UnitY_vec3);
    else if (axis == "Z")
      return Eigen::AngleAxis<real>(angle, UnitZ_vec3);
    else
      ACME_ERROR("acme::rotate(): invalid axis string.")
  }

  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -

  real
  angle(
      vec3 const &input0,
      vec3 const &input1)
  {
    return acme::acos(input0.dot(input1) / (input0.norm() * input1.norm()));
  }

  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -

  void
  eulerAngles(
      mat3 const &rotation,
      integer i,
      integer j,
      integer k,
      vec3 &angles)
  {
    vec3 angles_tmp(rotation.eulerAngles(i, j, k));
    angles[i] = angles_tmp[0];
    angles[j] = angles_tmp[1];
    angles[k] = angles_tmp[2];
  }
