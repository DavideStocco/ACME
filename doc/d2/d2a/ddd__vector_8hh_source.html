389  inline const vector<T> &orthogonalVector(void)
390  const
391  {
392  if (abs(this->x()) <= abs(this->y()) && abs(this->x()) <= abs(this->z()))
393  {
394  return new vector<T>(0, this->z(), -this->y());
395  }
396  else if (abs(this->y()) <= abs(this->x()) && abs(this->y()) <= abs(this->z()))
397  {
398  return new vector<T>(this->z(), 0, -this->x());
399  }
400  else
401  {
402  return new vector<T>(this->y(), -this->x(), 0);
403  }
404  }
